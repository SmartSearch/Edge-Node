change
Fires when the grid selection has changed.

Example

 $("#grid").kendoGrid({
     change: function(e) {
         // handle event
     }
 });

To set after initialization

 // get a reference to the grid
 var grid = $("#grid").data("kendoGrid");
 // bind to the change event
 grid.bind("change", function(e) {
     // handle event
 }
